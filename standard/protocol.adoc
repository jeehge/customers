== SMS 전송 요청 (PC -> Server)
[source, json]
----
{
  "server": {
    "type": "OUT_SMS",
    "receivers": [
      {
        "mobile": "010-1111-2222"
      },
      {
        "mobile": "010-3333-4444"
      }
    ]
  },
  "client": {
    "temp-id": "xxxxxx",
    "text": "SMS 문자 내용"
  }
}
----

== SMS/MMS 전송 결과 (Android -> Server)
[source, json]
----
{
  "server": {
    "type": "OUT_SMS_MMS_RESULT",
    "receivers": [
      {
        "mobile": "010-1111-2222",
        "name": "홍길동1"
      },
      {
        "mobile": "010-1111-2223",
        "name": "홍길동2"
      },
      {
        "mobile": "010-1111-2224",
        "name": ""
      }
    ]
  },
  "client": {
    "sub-type": "SMS/MMS",
    "temp-id": "xxxxxx",
    "sms": [
      {
        "receiver": "010-1111-2222",
        "result": true,
        "room-id": 222,
        "message-id": 11,
        "unread-count": 1
      },
      {
        "receiver": "010-1111-2222",
        "result": false,
        "room-id": 223,
        "message-id": 22,
        "unread-count": 1
      }
    ],
    "mms": {
      "result": true,
      "room-id": 222,
      "message-id": 11,
      "unread-count": 11
    }
  }
}
----
